```sql
-- Who calls RollbackTo?
SELECT called_by FROM code_units WHERE method_name='RollbackTo';
-- Result: (empty вҖ” no other method in Dom_Graph calls it directly)
--         it's invoked from use_memunit.py at the script level, not via a method call

-- What does RollbackEngine.RollbackTo call?
SELECT calls FROM code_units
WHERE class_name='RollbackEngine' AND method_name='RollbackTo';
-- Result: Run,_p,dumps,int,isoformat,items,len,str,utcnow

-- Dead methods (no callers) вҖ” 228 of 1,801 methods have no recorded caller
SELECT COUNT(*) FROM code_units
WHERE unit_type='METHOD' AND (called_by IS NULL OR called_by='');

-- Most connected methods (highest change impact)
SELECT class_name, method_name, LENGTH(calls) as call_count
FROM code_units WHERE unit_type='METHOD'
ORDER BY call_count DESC LIMIT 10;

-- VBStyle compliance check
SELECT class_name, method_name, return_type FROM code_units
WHERE is_vbstyle=1 AND unit_type='METHOD' AND return_type!='Tuple3';
```
